Count Characters

        const countCharacters = (S) => 
        { 
          let countA=0;
          let countD=0;
          let arr=[];
        
          for(let i=0; i< S.length; i++){
            if(S.charAt(i)==='A'){
             countA++;
            } 
            else if(S.charAt(i)==='D'){
              countD++;
            }
          }
          arr=[countA , countD];
          return arr;
        };

Count the Heads

        var Count_Occ = (s) => 
        {
            let n=s.length;
            let str=s.split('').sort().join('');
            let ans='';
            
            for(let i=0;i< n; i++){
              let count=0;
              for(let j=0;j< n; j++){
                if(str[i]===str[j]&&i>j){
                  break;
                }
                if(str[i]===str[j]){
                  count+=1;
                }
              }
              if(count>1){
                ans+=str[i]+count;
              }
            }
            return ans;
        };

Find the Winner

        var Game_Winner = (S) => 
        {
         let countA=0;
         let countD=0;
          for(let i=0; i< S.length; i++){
             let c=S.charAt(i);
            if(c==='A'){
              countA++;
            }
            else if(c==='D'){
              countD++;
            }
            
          }
          if(countA > countD){
            return "Aditya";
          }
          else if(countA < countD){
            return "Danish";
          }
          else{
            return "Draw";
          }
        }

Plaindrome Check

        var Palin_Check = (str) =>
        {
              var len = str.length;
          for (var i = 0; i < len/2; i++) {
            if (str[i] !== str[len - 1 - i]) {
              return "False";
            }
          }
          return "True";
        }
